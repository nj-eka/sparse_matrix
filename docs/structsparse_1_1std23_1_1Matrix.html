Detailed Description

template<std::copyable T, size_t N_DIMS>
struct sparse::std23::Matrix< T, N_DIMS >

sparse matrix with N_DIMS "infinite" dimensions

Matrix<int, 3> m(-1);
m[100, 200, 300] = 1;
assert(m[100, 200, 300] == 1);
assert(m[1, 2, 3] == -1);
assert(m.size() == 1);
m[100, 200, 300] = -1;
assert(m.size() == 0);
